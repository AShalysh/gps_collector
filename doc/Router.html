def resolve(env)
  method = env['REQUEST_METHOD']
  path = env['PATH_INFO']
  key = "#{method} #{path}"

  if routes.key?(key)
    value = routes[key]
    create_controller(value).call
  else
    Controller.new.not_found
  end
rescue Exception => e
  puts e.message
  puts e.backtrace
  Controller.new.internal_error
end
